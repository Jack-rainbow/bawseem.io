# new的过程中操作了什么？

    var a = new Person('x','1');
    new Person  {
        var obj = {};
        //创建一个对象
        obj._proto_ = Person.prototype; 
        // 将新创建的空对象的隐式原型指向其构造函数的显示原型。
        var result  = Person.call(obj,arguments);
        //Person继承obj上边的参数也就是实参，并且改变this指向(因为call，apply，bind具有改变this的功能)
        return typeof result === 'obj' ? result  :  obj;
        //如果返回值是一个新对象的话那么直接直接返回该对象否则就返回形参
    }
