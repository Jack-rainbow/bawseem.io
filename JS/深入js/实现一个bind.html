    # Array.prototype.slice.apply(argumens,[1])的作用

    目的是拿到参数里边除第一个以外后面的所有参数(执行函数时传递的参数)

    # Array.prototype.slice.apply(arguments,[0])的作用

    返回函数执行的时候传递的一系列参数

备注：只有两者合并了之后才是返回函数的完整参数

    # 手动实现一个call

    Function.prototype.myCall = function (obj, ...arg) {
    let val;
    if(obj === null || obj === undefined){
        obj = window;
    } else {
        obj = Object(obj);
    }
    obj._fn = this;
    obj._fn();
    val = obj._fn(...arg);
    delete obj._fn;
    return val;
}
let o = {
    name: 'o',
    fn: function () {
        console.log(this.name);
    }
}
let test = {
    name: 'test'
}
o.fn();
o.fn.call(test);
o.fn.myCall(test);

    var _this = this;
    /* Array.prototype.slice.apply(arguments,[1]);
    * 拿到参数传递进来第一个以外的所有参数
    */
    let args = Array.prototype.slice.apply(arguments,[1]);
    fNOP = function () {},
    bound = function(){
        /* Array.prototype.slice.apply(arguments,[0]);
        * 返回函数执行时传递进来的一系列参数
        */
        let args2 = Array.from(arguments);
        return _this.apply(that || window, 
        args.concat(args2)
        );
    }
    fNOP.prototype = _this.prototype;
    
    bound.prototype = new fNOP();
    
    return bound;
}
