  [Symbol.toPrimitive](hint) {

//变量为对象
let obj = {
    valueOf:()=>{
        return 'object'
    },
    toString:()=>{
        return 'toString'
    }
}
obj + 1 //object1

//变量为字符串
let str = String({
    valueOf:()=>{
        return 'object'
    },
    toString:()=>{
        return 'toString'
    }
})
str + 1 // toString1

3.比较运算


1.==号两边有NaN，返回false
2.==号左边如果是null或者undefined，返回true(自身与自身相等 null==null//undefined == undefined)，否则返回false
3.如果==号两边类型不一致，且x，y为String,Number,Boolean中的某一类型的时候，则将X,Y使用toNumber()进行转换为Number然后在比较
4.如果x,y有一个是object,则优先使用toPrimtive函数转换为原始类型，然后在进行比较
