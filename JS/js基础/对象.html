# hasOwnProperty

1.小记

查看一个对象中是否有某个属性

2.实现方法

Object.prototype.hasOwnProperty = function (val) {
    let _this = this;
    for (i in _this) {
        if (i == val) {
            return true;
        }
    }
    return false;
}

# is

1.小记 判断两个参数是否相等 2.实现方法

Object.prototype.is = function(val1,val2){
    if(val1 === val2){
        return false
    }else {
        return val1 !== val1 && val2 !== val2
    }
    return true;
}

# values

1.小记

返回对象中的属性参数

2.实现方法

Object.prototype.values2 = function (val) {
    let arr = []
    for (i in val) {
        if (typeof val[i] != 'function')
            arr.push(val[i].toString())
    }
    return arr;
}
