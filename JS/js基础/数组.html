# every

1.小记

查看数组中个条件是否都满足函数中的判断

2.实现方法

Array.prototype.every = function (fun) {
    let _this = Object(this);
    let thisArg = arguments.length >= 2 ? arguments[1] : undefined;
    if (typeof fun !== 'function') {
        throw new TypeError();
    }
    for (var i = 0; i < _this.length; i++) {
        if (i in _this && !fun.call(thisArg, _this[i], i, _this))   //主要作用在这里
            return false;
    }
    return true;
}

Array.prototype.join = function (val) {
    let str = '';
    if (arguments.length == 0) {
        return this.toString();
    } else {
        this.forEach((ele, i, arr) => {
            str += val + ele.toString();
        })
    }
    return str;
}

# Array.of

1.小记

返回数组且不考虑参数的数量及类型

2.实现方法

Array.of = function (val) {
    let checkoutType = Object.prototype.toString.call(val);
    if (checkoutType == '[object Object]') {
        return [].push(Object.assign(val).valueOf());
    } else {
        let setVal = [];
        setVal[0] = val;
        return setVal;
    }
}

Array.prototype.find = function (fun) {
    if (this == null && typeof fun !== 'function') {
        throw new TypeError();
    }
    let k = 0;
    let _this = Object(this);
    let thisArg = arguments.length >= 2 ? arguments[1] : void 0;
    let len = _this.length >>> 0;
    while (k < len) {
        if (k in _this) {
            if (fun.call(thisArg, _this[k], k, _this)) {
                return _this[k];
            }
        }
        k++;
    }
    return undefined;
}
