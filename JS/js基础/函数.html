# call

  • 作用

1.修改this执行 2.调用函数(自执行) 3.第二参数可以多个参数

  • 如何实现call?
Function.prototype.Call = function (obj, ...arg) {
  let val;
  if(obj === null || obj === undefined){
      obj = window;
  } else {
      obj = Object(obj);
  }
  obj._fn = this;
  obj._fn();
  val = obj._fn(...arg);
  delete obj._fn;
  return val;
}
