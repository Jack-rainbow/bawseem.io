# 代理模式

    let a = {
        name: 'yyy',
        age: '18',
        phone:'1888888888'
    }
    let b = new Proxy(a,{
        get:function(target,key){
            //target 原对象
            //key 键值
            console.log(target,key)
        },
        set:function(target,key,value){
            if(key == 'customPrice'){
                if(value < 100){
                    throw new Error('过低');
                }
            }
             console.log(value)
        }
    })
    a.name = '123';
    console.log(b.name)

代理模式、适配器模式、装饰器模式区别

  • 适配器只提供一个接口，代理模式提供一模一样的接口，并且接口不可以改动。
  • 装饰器模式:原有功能不变，在他基础上扩展,代理模式显示原有功能，但是必须通过中间件去访问
