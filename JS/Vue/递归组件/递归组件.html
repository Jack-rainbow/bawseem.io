findComponentAllChilren = (content:Array<object>,componentName:string,exceptMe = true)=>{
    let componentArr:Array<object> = [];
    let childrenList = content.$parent.$children.filter((item:object)=>{item.$options.name === componentName });

    if(exceptMe){
        childrenList.splice(res,1);
    }
    return componentArr;
}

2.如何解决递归中的emit事件?

/**
 * @description: 从上向下传递事件(父组件-子组件)
 * @param {componentName}  组件名字
 * @param {eventName}  自定义单击事件名字
 * @param {params}  传递的数据
 */
brodcast = (componentName:String,eventName:String,params:any):void=>{
    this.$children.forEach(child=>{
        const name:String = child.$options.name;
        if(name === componentName){
            child.$emit.apply(child,[eventName,...params]);
        }else{
            brodcast.apply(child,[componentName,eventName,[...params]]);
        }
    })
}
/**
 * @description: 从下向上传递事件(子组件向父组件传递)
 * @param {componentName}  组件名字
 * @param {eventName}  自定义单击事件名字
 * @param {params}  传递的数据
 */
dispatch = (componentName:String,eventName:String,params:any):void=>{
    let parent:Array = this.$parent || this.$root; 
    let name:String = parent.$options.name;
    // 通过获取的parent.name去判断当前的name是否等于最上级的parent(外部引用)
    while(parent && (name || name !== componentName)){
        parent:Array = parent.$parent;
        if(parent){
            name:String = parent.$options.name;
        }
    }
    if(parent){
        parent.$emit.apply(parent,[eventName,[...params]])
    }
}
export default{
    methods:{
        dispatch,
        broadcast(componentName, eventName, params) {
            broadcast.call(this, componentName, eventName, params);
        }
    }
}
