# 关于work-break与float之间联系

 .a{
        word-break: break-all;
    }
    .a span:last-child{
        float: right;
    }
    .b{
        display: flex;
        flex-direction: column;
    }
 <ul class='b'> 
    <li class='a '>
        <span>1111111111111111111111111111111111111111111111111</span>
        <span>2018-12-18</span>
    </li> 
    <li class='a '>
        <span>111111111111111111111111111111111</span>
        <span>2018-12-18</span>
    </li> 
</ul>
